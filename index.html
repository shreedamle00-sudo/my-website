# RAVDESS Emotion Recognition - Complete Training Pipeline

import os
import numpy as np
import librosa
import joblib
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import LabelEncoder, StandardScaler
from sklearn.neural_network import MLPClassifier
from sklearn.metrics import classification_report, confusion_matrix
import glob

# Configuration
DATASET_PATH = r"D:\VS_Code_DS_Python\ML projest 3rd year\dataset\ALL"
MODEL_SAVE_PATH = "emotion_model.pkl"
SAMPLE_RATE = 22050

# RAVDESS emotion mapping (filename format: 03-01-{emotion}-...)
EMOTION_MAP = {
    '01': 'neutral', '02': 'calm', '03': 'happy', '04': 'sad',
    '05': 'angry', '06': 'fearful', '07': 'disgust', '08': 'surprised'
}

def extract_features(file_path):
    """Extract 60-dimensional feature vector from audio file"""
    try:
        # Load audio: 22.05kHz, 3 seconds max
        y, sr = librosa.load(file_path, sr=SAMPLE_RATE, duration=3)
        
        # Preprocessing: trim silence (top_db=20) and normalize
        y, _ = librosa.effects.trim(y, top_db=20)
        y = librosa.util.normalize(y)
        
        # Feature extraction (60 total features)
        mfccs = np.mean(librosa.feature.mfcc(y=y, sr=sr, n_mfcc=40).T, axis=0)           # 40
        chroma = np.mean(librosa.feature.chroma_stft(y=y, sr=sr).T, axis=0)              # 12  
        zcr = np.mean(librosa.feature.zero_crossing_rate(y).T, axis=0)                   # 1
        spec_contrast = np.mean(librosa.feature.spectral_contrast(y=y, sr=sr).T, axis=0) # 7
        
        return np.hstack([mfccs, chroma, zcr, spec_contrast])
    except Exception as e:
        print(f"Error processing {file_path}: {e}")
        return None

def get_emotion_from_filename(filename):
    """Extract emotion from RAVDESS filename: 03-01-{emotion}-..."""
    parts = os.path.basename(filename).split('-')
    if len(parts) >= 3:
        return EMOTION_MAP.get(parts[2], None)
    return None

# Model Configuration (Exact hyperparameters used)
model = MLPClassifier(
    hidden_layer_sizes=(256, 128, 64),    # 3 hidden layers
    activation='relu',                     # ReLU activation
    solver='adam',                         # Adam optimizer
    alpha=0.0001,                         # L2 regularization
    learning_rate_init=0.001,             # Initial learning rate
    max_iter=500,                         # Maximum epochs
    early_stopping=True,                  # Early stopping enabled
    validation_fraction=0.1,              # 10% for validation
    n_iter_no_change=10,                  # Patience for early stopping
    tol=0.0001,                          # Tolerance for optimization
    random_state=42,                      # Reproducibility
    verbose=True                          # Training progress
)
